Het is heel gebruikelijk dat dezelfde objecten in een lijst staan.
Maak een lijst aan en voeg 4 Club objecten toe aan de lijst.
Itereer over de lijst heen en print de verschillende clubnamen uit op jouw pagina.

const Clubinfo = function (name, sport, members) {
this.name = name;
this.sport = sport;
this.members = members;
};

const ajax = new Clubinfo("ajax", "football", 1000);
const manchester = new Clubinfo("manchester", "football", 2000);
const psv = new Clubinfo("psv", "football", 12345);
const vfn = new Clubinfo("vfn", "volleyball", 23453);

// make a list with similar objects

let clubCluster = [ajax, manchester, psv, vfn];

// loop through object properties

for (i = 0; i < clubCluster.length; i++) {
console.log(clubCluster[i].name); }
